function mdc(num1, num2) {

var resto;

do {
resto = num1 % num2;

num1 = num2;
num2 = resto;

} while (resto != 0);

document.write("Máximo divisor comum: " + num1);
}

var n1 = parseInt(prompt("Digite o primeiro número: "));
var n2 = parseInt(prompt("Digite o segundo número:"));

mdc(n1,n2);
